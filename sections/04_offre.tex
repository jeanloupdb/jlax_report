\end{pointcle}

\subsection{Outils à éviter}

\begin{table}[htbp]
  \centering
  \caption{Outils exclus du périmètre}
  \renewcommand{\arraystretch}{1.4}
  \begin{tabularx}{\textwidth}{>{\bfseries}p{3cm} X}
    \toprule
    \rowcolor{tabheader}

    \rowcolor{rowgray}
    Azure Sentinel & Expertise longue, coût explosif selon volume de logs \\
    Power BI & Interface complexe, formation requise → Metabase ou Grafana \\
    \bottomrule
  \end{tabularx}
\end{table}
